## Aide

### Forward ref

```js
import React from 'react'

export default React.forwardRef(function MyInput(props, ref) {
  return <input ref={ref} {...props} />
})
```

### Utilisation d'une ref

```js
function Example() {
  const inputRef = useRef()

  useEffect(() => {
    inputRef.current.focus()
  }, [])

  return <input ref={inputRef} />
```
